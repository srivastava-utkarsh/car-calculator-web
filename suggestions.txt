        function calculateEMI(principal, rate, tenure) {
            const monthlyRate = rate / (12 * 100);
            const emi = (principal * monthlyRate * Math.pow(1 + monthlyRate, tenure)) / 
                       (Math.pow(1 + monthlyRate, tenure) - 1);
            return emi;
        }

        function formatCurrency(amount) {
            return '‚Çπ' + Math.round(amount).toLocaleString('en-IN');
        }

        function formatTenure(months) {
            const years = Math.floor(months / 12);
            const remainingMonths = months % 12;
            
            if (years === 0) {
                return `${remainingMonths} month${remainingMonths !== 1 ? 's' : ''}`;
            } else if (remainingMonths === 0) {
                return `${years} year${years !== 1 ? 's' : ''}`;
            } else {
                return `${years} year${years !== 1 ? 's' : ''} ${remainingMonths} month${remainingMonths !== 1 ? 's' : ''}`;
            }
        }

        function calculatePrepayment() {
            const loanAmount = parseFloat(document.getElementById('loanAmount').value);
            const interestRate = parseFloat(document.getElementById('interestRate').value);
            const loanTenure = parseInt(document.getElementById('loanTenure').value);
            const prepaymentAmount = parseFloat(document.getElementById('prepaymentAmount').value);
            const paymentFrequency = document.getElementById('paymentFrequency').value;
            const prepaymentPenalty = parseFloat(document.getElementById('prepaymentPenalty').value);

            // Calculate original loan details
            const originalEMI = calculateEMI(loanAmount, interestRate, loanTenure);
            const originalTotalPayment = originalEMI * loanTenure;
            const originalTotalInterest = originalTotalPayment - loanAmount;

            // Calculate prepayment frequency multiplier
            let prepaymentFrequencyMultiplier;
            switch(paymentFrequency) {
                case 'monthly': prepaymentFrequencyMultiplier = 12; break;
                case 'quarterly': prepaymentFrequencyMultiplier = 4; break;
                case 'half-yearly': prepaymentFrequencyMultiplier = 2; break;
                case 'yearly': 
                default: prepaymentFrequencyMultiplier = 1; break;
            }

            // Simulate loan with prepayments
            let remainingPrincipal = loanAmount;
            let month = 0;
            let totalInterestPaid = 0;
            let totalPrepaymentPaid = 0;
            const monthlyRate = interestRate / (12 * 100);

            while (remainingPrincipal > 0.01 && month < loanTenure) {
                month++;
                
                // Calculate monthly interest
                const monthlyInterest = remainingPrincipal * monthlyRate;
                const monthlyPrincipal = originalEMI - monthlyInterest;
                
                totalInterestPaid += monthlyInterest;
                remainingPrincipal -= monthlyPrincipal;

                // Apply prepayment based on frequency
                if (month % (12 / prepaymentFrequencyMultiplier) === 0) {
                    const prepaymentWithPenalty = prepaymentAmount * (1 + prepaymentPenalty / 100);
                    const actualPrepayment = Math.min(prepaymentWithPenalty, remainingPrincipal);
                    remainingPrincipal -= actualPrepayment;
                    totalPrepaymentPaid += actualPrepayment;
                }
            }

            const newTenure = month;
            const newTotalInterest = totalInterestPaid;
            const interestSavings = originalTotalInterest - newTotalInterest;
            const timeSavings = loanTenure - newTenure;

            // Update UI
            document.getElementById('originalTenure').textContent = formatTenure(loanTenure);
            document.getElementById('originalInterest').textContent = formatCurrency(originalTotalInterest);
            document.getElementById('originalEMI').textContent = formatCurrency(originalEMI);

            document.getElementById('newTenure').textContent = formatTenure(newTenure);
            document.getElementById('newInterest').textContent = formatCurrency(newTotalInterest);
            document.getElementById('newEMI').textContent = formatCurrency(originalEMI);

            document.getElementById('totalSavings').textContent = formatCurrency(interestSavings);
            document.getElementById('timeSaved').textContent = formatTenure(timeSavings);

            // Update detailed table
            document.getElementById('beforeTenure').textContent = formatTenure(loanTenure);
            document.getElementById('afterTenure').textContent = formatTenure(newTenure);
            document.getElementById('tenureSaved').textContent = formatTenure(timeSavings);

            document.getElementById('beforeInterest').textContent = formatCurrency(originalTotalInterest);
            document.getElementById('afterInterest').textContent = formatCurrency(newTotalInterest);
            document.getElementById('interestSaved').textContent = formatCurrency(interestSavings);

            document.getElementById('beforeEMI').textContent = formatCurrency(originalEMI);
            document.getElementById('afterEMI').textContent = formatCurrency(originalEMI);

            document.getElementById('beforeTotal').textContent = formatCurrency(originalTotalPayment);
            document.getElementById('afterTotal').textContent = formatCurrency(newTotalInterest + loanAmount + totalPrepaymentPaid);
            document.getElementById('totalAmountSaved').textContent = formatCurrency(interestSavings);

            // Show results section
            document.getElementById('resultsSection').style.display = 'block';
            
            // Update step indicator
            const steps = document.querySelectorAll('.step');
            steps.forEach((step, index) => {
                if (index <= 2) {
                    step.classList.add('active');
                }
            });

            // Smooth scroll to results
            document.getElementById('resultsSection').scrollIntoView({ 
                behavior: 'smooth',
                block: 'start'
            });
        }

        // Initialize the calculator
        document.addEventListener('DOMContentLoaded', function() {
            syncInputs();
            // Calculate with default values
            setTimeout(calculatePrepayment, 500);
        });
